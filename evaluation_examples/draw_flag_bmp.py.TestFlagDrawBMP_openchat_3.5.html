#include <stdio.h>
#include <stdlib.h>
#include <png.h>

#define WIDTH 16
#define HEIGHT 9
#define DEPTH 8

int main() {
    // Create a memory buffer to hold the image data
    unsigned char *image_data = malloc(WIDTH * HEIGHT * DEPTH);

    // Initialize the PNG structure
    png_structp png_ptr = png_create_write_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
    if (!png_ptr) {
        fprintf(stderr, "Error creating PNG write struct\n");
        return 1;
    }

    // Initialize the info structure
    png_infop info_ptr = png_create_info_struct(png_ptr);
    if (!info_ptr) {
        fprintf(stderr, "Error creating PNG info struct\n");
        return 1;
    }

    // Set up error handling
    if (setjmp(png_jmpbuf(png_ptr))) {
        fprintf(stderr, "Error writing PNG file\n");
        return 1;
    }

    // Open the memory buffer as a file
    FILE *fp = fopen(NULL, "wb");
    if (!fp) {
        fprintf(stderr, "Error opening memory buffer as file\n");
        return 1;
    }

    // Set up the write function
    png_set_write_fn(png_ptr, fp, write_data, NULL);

    // Write the header
    png_init_io(png_ptr, info_ptr);
    png_set_IHDR(png_ptr, info_ptr, WIDTH, HEIGHT, DEPTH,
                  PNG_COLOR_TYPE_RGB, PNG_INTERLACE_NONE,
                  PNG_COMPRESSION_TYPE_DEFAULT, PNG_FILTER_TYPE_DEFAULT);
    png_write_info(png_ptr, info_ptr);

    // Write the image data
    for (int y = 0; y < HEIGHT; y++) {
        for (int x = 0; x < WIDTH; x++) {
            int index = y * WIDTH + x;
            if (x < 7 || (x == 7 && y < 3) || (x == 13 && y < 3) || (x == 19 && y < 3) || y == 5) {
                image_data[index * DEPTH] = 255;
                image_data[index * DEPTH + 1] = 0;
                image_data[index * DEPTH + 2] = 0;
            } else {
                image_data[index * DEPTH] = 0;
                image_data[index * DEPTH + 1] = 0;
                image_data[index * DEPTH + 2] = 255;
            }
        }
    }
    png_write_row(png_ptr, image_data);
    png_write_end(png_ptr, NULL);

    // Clean up
    png_destroy_write_struct(&png_ptr, &info_ptr);
    fclose(fp);
    free(image_data);

    return 0;
}

int write_data(png_structp png_ptr, png_bytep data, png_size_t length) {
    fwrite(data, 1, length, stdout);
    return PNG_ERROR_NO_ERROR;
}
