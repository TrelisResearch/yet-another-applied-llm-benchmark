def evaluate(program):
    registers = {f'R{i}': 0 for i in range(1, 9)}
    flags = {'flag': False}
    memory = {i: 0 for i in range(100)}
    instruction_pointer = 0

    def get_value(arg):
        if arg.isdigit():
            return int(arg)
        elif arg.startswith('R'):
            return registers[arg]
        else:
            return memory[int(arg)]

    def set_value(arg, value):
        if arg.startswith('R'):
            registers[arg] = value
        else:
            memory[int(arg)] = value

    program = program.split('\n')
    while instruction_pointer < len(program):
        line = program[instruction_pointer].strip()
        if line == '' or line.startswith('//'):
            instruction_pointer += 1
            continue

        parts = line.split()
        op = parts[0]
        if op == 'SET':
            registers[parts[1]] = get_value(parts[2])
        elif op in ['ADD', 'SUB', 'MUL', 'DIV', 'MOD']:
            arg1 = get_value(parts[2])
            arg2 = get_value(parts[3])
            if op == 'ADD':
                result = arg1 + arg2
            elif op == 'SUB':
                result = arg1 - arg2
            elif op == 'MUL':
                result = arg1 * arg2
            elif op == 'DIV':
                result = arg1 // arg2
            elif op == 'MOD':
                result = arg1 % arg2
            registers[parts[1]] = result
        elif op in ['EQ', 'NEQ', 'LT', 'LTE', 'GT', 'GTE']:
            arg1 = get_value(parts[1])
            arg2 = get_value(parts[2])
            if op == 'EQ':
                flags['flag'] = arg1 == arg2
            elif op == 'NEQ':
                flags['flag'] = arg1 != arg2
            elif op == 'LT':
                flags['flag'] = arg1 < arg2
            elif op == 'LTE':
                flags['flag'] = arg1 <= arg2
            elif op == 'GT':
                flags['flag'] = arg1 > arg2
            elif op == 'GTE':
                flags['flag'] = arg1 >= arg2
        elif op == 'INC':
            registers[parts[1]] += 1
        elif op == 'DEC':
            registers[parts[1]] -= 1
        elif op == 'JMP':
            instruction_pointer = program.index(parts[1] + ':')
            continue
        elif op == 'JT':
            if flags['flag']:
                instruction_pointer = program.index(parts[1] + ':')
                continue
        elif op == 'JF':
            if not flags['flag']:
                instruction_pointer = program.index(parts[1] + ':')
                continue
        elif op == 'LOAD':
            registers[parts[1]] = memory[get_value(parts[2])]
        elif op == 'STORE':
            memory[get_value(parts[2])] = registers[parts[1]]
        elif op == 'HCF':
            break

        instruction_pointer += 1

    return [memory[i] for i in range(20)]
