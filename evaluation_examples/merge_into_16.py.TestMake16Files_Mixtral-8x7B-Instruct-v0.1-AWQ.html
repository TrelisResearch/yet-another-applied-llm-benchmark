import shutil

# Read in the list of file paths from the file
with open('/tmp/files', 'r') as f:
    file_paths = [line.strip() for line in f]

# Create the merged files
for i in range(16):
    with open('/tmp/merged_{}.txt'.format(i), 'w') as outfile:
        for j in range(len(file_paths) // 16 if i < len(file_paths) % 16 else len(file_paths) % 16):
            try:
                with open(file_paths.pop(0), 'r') as infile:
                    shutil.copyfileobj(infile, outfile)
            except IOError as e:
                print('Error processing file {}: {}'.format(file_paths.pop(0), e))
