class BertLMHead(torch.nn.Module):
    def __init__(self, hidden_size, num_labels):
        super(BertLMHead, self).__init__()
        self.bert = base_model_name
        self.classifier = torch.nn.Linear(hidden_size, num_labels)

    def forward(self, inputs_embeds, segment_ids, masked_ids, masked_positions, masked_weights,):
        cls_representation, _ = self.bert(
            inputs_embeds,
            token_type_ids=segment_ids,
            attention_mask=inputs_embeds.ne(tokenizer.pad_token_id),
            masked_positions=masked_positions,
        )

        cls_representation = self.dropout(cls_representation[:, 0, :])
        logits = self.lm_head(cls_representation)
        return logits
