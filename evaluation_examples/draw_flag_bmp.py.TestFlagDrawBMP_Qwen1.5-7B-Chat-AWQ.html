#include <stdio.h>
#include <stdlib.h>

// BMP header structure
typedef struct {
    uint16_t signature;        // 0x4D42 (BM)
    uint32_t fileSize;        // Size of the file in bytes
    uint16_t reserved1;       // Always 0
    uint16_t reserved2;       // Always 0
    uint32_t dataOffset;      // Offset to the pixel data
    uint32_t headerSize;      // Size of the header in bytes
    uint32_t width;           // Flag width
    uint32_t height;          // Flag height
    uint16_t planes;          // 1 (24-bit color)
    uint16_t bitsPerPixel;    // 24
    uint32_t compression;     // 0 (No compression)
    uint32_t imageSize;       // Size of the image data
    uint32_t xPixelsPerMeter; // Always 1
    uint32_t yPixelsPerMeter; // Always 1
    uint32_t colorsUsed;      // 256 (24-bit color)
    uint32_t colorsImportant; // Always 0
} BMPHeader;

// RGB values for the American flag
const uint8_t flag_colors[] = {
    255, 0, 0, // Red
    255, 63, 0, // White (top left)
    0, 0, 255, // Blue
    0, 63, 255, // White (top right)
    0, 255, 0, // Red
    0, 255, 63, // White (bottom left)
    255, 0, 255 // Blue
};

// Function to create a BMP file
void create_flag_bmp() {
    BMPHeader header;
    header.signature = 0x4D42; // BMP signature
    header.fileSize = sizeof(header) + sizeof(flag_colors) * 50 * 50; // Size of the header and pixel data
    header.dataOffset = sizeof(header);
    header.headerSize = 54; // BMP header size
    header.width = 50; // Flag width
    header.height = 50; // Flag height
    header.planes = 1;
    header.bitsPerPixel = 24;
    header.compression = 0;
    header.imageSize = header.width * header.height * 3; // 3 bytes per pixel (RGB)
    header.colorsUsed = 256; // 24-bit color
    header.colorsImportant = 0;

    // Allocate memory for the pixel data
    uint8_t* pixels = malloc(header.imageSize);
    if (!pixels) {
        printf("Error: Failed to allocate memory for the BMP file.\n");
        return;
    }

    // Fill the pixel data with flag colors
    for (int y = 0; y < header.height; ++y) {
        for (int x = 0; x < header.width; ++x) {
            int flag_x = x * 3; // Convert to pixel coordinates
            int flag_y = header.height - y - 1; // Reverse the order (bottom-up)

            // Set the flag colors
            pixels[flag_x] = flag_colors[flag_y * 4];
            pixels[flag_x + 1] = flag_colors[flag_y * 4 + 1];
            pixels[flag_x + 2] = flag_colors[flag_y * 4 + 2];
        }
    }

    // Write the BMP header and pixel data to stdout
    printf("BMP file:\n");
    printf("Size: %u bytes\n", header.fileSize);
    printf("Width: %u\n", header.width);
    printf("Height: %u\n", header.height);
    printf("Color planes: 1\n");
    printf("Bits per pixel: 24\n");
    printf("Compression: 0\n");
    printf("Image size: %u\n", header.imageSize);
    printf("Colors: 256\n");

    // Write the header
    for (uint16_t i = 0; i < sizeof(header); ++i) {
        printf("%04X ", header.dataOffset + i);
    }
    printf("\n");

    // Write the pixel data
    for (int i = 0; i < header.imageSize; ++i) {
        printf("%02X ", pixels[i]);
    }
    printf("\n");

    // Free the pixel data
    free(pixels);

    printf("Flag BMP created.\n");
}

int main() {
    create_flag_bmp();
    return 0;
}
