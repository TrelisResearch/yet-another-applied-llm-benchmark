@lru_cache(maxsize=None)
def solve_dp(graph, lines_left, distance_since_last):
    # Initialize the result
    result = [(10**9, [])] * (len(graph)+1)

    for i in reversed(range(len(graph))):
        for remaining_lines in reversed(range(lines_left+1)):
            distance = (1e9, [])

            if remaining_lines == 0 and i == len(graph):
                distance = (0, [])
                continue

            if remaining_lines > 0:
                distance2 = result[i+1][remaining_lines-1]
                distance2 = (distance2[0]+(i-8)**2, distance2[1]+[i])

            distance = min(distance, distance2,
                          (1e9, []))

            distance = (distance[0]+(i-distance_since_last)**2,
                       distance[1]+[i])

            result[i] = distance

    return result[0]
