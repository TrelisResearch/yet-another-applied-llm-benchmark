    float dp[n + 1][lines_left + 1][10];
    int how[n + 1][lines_left + 1][10];

    for (int i = n; i >= 0; i--) {
        for (int j = 0; j <= lines_left; j++) {
            for (int k = 0; k < 10; k++) {
                if (i == n && j == 0) {
                    dp[i][j][k] = 0;

                    if (i < n && k < 9) {
