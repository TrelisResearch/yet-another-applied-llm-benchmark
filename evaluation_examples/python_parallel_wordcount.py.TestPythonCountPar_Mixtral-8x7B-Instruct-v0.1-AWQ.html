#!/usr/bin/env python3

import json
import concurrent.futures
import argparse
import sys

parser = argparse.ArgumentParser(description="Parallel word-counting program")

args = parser.parse_args()

def word_count(start_line: int) -> dict:
    result = {}
    with open("/tmp/file.txt", "r") as f:
        f.seek(start_line)
        for line in iter(lambda: f.read(1024), b""):
            for word in line.split():
                word = word.decode("utf-8").rstrip(",\"\'[]")
                result[word] = result.get(word, 0) + 1
    return result

if __name__ == "__main__":
    with concurrent.futures.ProcessPoolExecutor(max_workers=4) as executor:
        for result in executor.map(word_count, range(0, 4 * (sys.getiove(), 1024), 4 * 1024)):
            print(json.dumps(result))
