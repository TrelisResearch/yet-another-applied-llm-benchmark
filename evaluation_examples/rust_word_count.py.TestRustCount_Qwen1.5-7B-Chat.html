fn main() {
    // Open the text file for reading
    let file = std::fs::File::open("/tmp/file.txt").expect("Failed to open file");

    // Read the entire file into a string
    let mut content = String::new();
    file.read_to_string(&mut content).expect("Failed to read file");

    // Split the content into words, ignoring punctuation
    let words = content
        .replace('.', '')
        .replace(',', '')
        .replace('!', '')
        .replace('?', '')
        .replace('-', '')
        .replace(';', '')
        .split_whitespace()
        .collect::<Vec<&str>>();

    // Count the occurrences of each word
    let word_count: Vec<(String, u32)> = words
        .into_iter()
        .enumerate()
        .map(|(i, word)| (word, 1))
        .collect();

    // Create a JSON object with word counts
    let json = json!({
        "word_count": word_count
    });

    // Print the JSON to stdout
    println!("{}", json);
}
