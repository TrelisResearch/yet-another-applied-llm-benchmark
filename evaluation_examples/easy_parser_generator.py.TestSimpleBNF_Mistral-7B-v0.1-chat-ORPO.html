import re

def evaluate(expression):
    # Define the regular expressions for each grammar rule
    expression_rule = r"(?P<expression>(?P<term>term)|(?P<expression>expression)(?P<operator>\+|-)(?P<term>term)))"
    term_rule = r"(?P<factor>factor)|(?P<term>term(?P<operator>\*\*|/|(?P<factor>factor)))"
    factor_rule = r"(?P<number>number)|(?P<factor>factor(?P<prefix_op>sin|cos))|(?P<factor>(?P<prefix_op>sin|cos)(?P<factor>factor)))"
    number_rule = r"(?P<digit>digit)|(?P<number>number(?P<digit>digit))"
    digit_rule = r"[0-9]"

    # Define the regular expression for the entire expression
    expression_regex = re.compile(expression_rule)

    # Evaluate the expression using the regular expressions
    match = expression_regex.search(expression)
    if match:
        expression = match.groupdict()
        term = expression.get("term")
        operator = expression.get("operator")
        factor = expression.get("factor")
        prefix_op = expression.get("prefix_op")
        number = expression.get("number")
        digit = expression.get("digit")

        # Evaluate the expression recursively
        if operator == "+":
            return evaluate(term) + evaluate(factor)
        elif operator == "-":
            return evaluate(term) - evaluate(factor)
        elif operator == "*":
            return evaluate(term) * evaluate(factor)
        elif operator == "/":
            return evaluate(term) / evaluate(factor)
        elif prefix_op == "sin":
            return math.sin(evaluate(factor))
        elif prefix_op == "cos":
            return math.cos(evaluate(factor))
        elif number:
            return int(number)
        elif digit:
            return int(digit)

    # Return the result of the evaluation
    return None
