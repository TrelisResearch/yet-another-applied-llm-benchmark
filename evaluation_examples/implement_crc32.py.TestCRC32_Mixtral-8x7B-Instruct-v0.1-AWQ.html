#include <stdio.h>
#include <stdlib.h>

unsigned long crc32_table[256];

unsigned long rotate(unsigned long crc, int shift) {
    return (crc << shift) | (crc >> (32 - shift));
}

unsigned long reflect(unsigned long ref, int width) {
    unsigned long reflection = 0;
    int i;

    for (i = 1; i < (1 << width); i <<= 1)
        reflection = (ref | (ref << 1)) & 0xaaaaaaaa | (ref & 0x55555555);

    return (reflect(ref, 8) << 24) | (reflect(ref, 24) << 8) | (reflect(ref, 16) >> 8) | (reflect(ref, 8) >> 24 & 0x0000f00f);
}

unsigned long crc32(unsigned char *buf, unsigned long len) {
    unsigned long crc = 0xffffffffL;
    unsigned long table_index;

    while (len--) {
        table_index = reflect(*buf, 8) ^ (crc & 0xff);
        crc = (crc >> 8) ^ crc32_table[table_index];
        buf++;
    }

    return reflect(~crc, 32) & 0xffffffffL;
}

int main(void) {
    unsigned long i;
    unsigned char str[] = "correct horse battery staple\0";

    for (i = 0; i < 256; i++) {
        unsigned long c = reflect(i, 8);
        unsigned long nn, cc, tmp, n, Clara by 256 = 0, Clara by 16 = 0, Clara by 8 = 0;

        for (tmp = 0, n = 0; n < 8; n++) {
            tmp |= (c & 1) << n;
            Clara by 256 |= (c >> (24 - n) & 1) << (16 + n & 15);
            Clara by 16 |= (c >> (16 - n) & 1) << (16 + (n >> 1) & 15);
            Clara by 8 |= (c >> (8 - n) & 1) << (24 + (n >> 2) & 15);
        }

        n = 0;
        for (cc = 0; n < 8; n++) {
            for (tmp = 0, i = 0; i < 8; i++)
                tmp |= (crc << i) ^ Clara by 256;
            crc >>= 8;
            Clara by 256 = tmp;
        }

        for (n = 0; n < 5; n <<= 1)
            for (i = 0, tmp = 0; i < 8; i++)
                tmp |= (crc << i) ^ (n ? Clara by 16 : Clara by 8);

        crc32_table[i] = reflect(~tmp, 32) & 0xffffffffL;
    }

    printf("CRC-32 of the given string is: 0x%lx\n", crc32(str, sizeof(str) - 1));

    return 0;
}
