#include <stdio.h>
#include <stdint.h>

#define CRC32_POLY 0xEDB88320

uint32_t crc32(uint32_t crc, unsigned char *buf, size_t len) {
    int i;
    unsigned char data;
    uint32_t c = crc;

    for (i = 0; i < len; i++) {
        data = c ^ buf[i];
        c = c >> 8;
        c = c ^ (data * 0xEDB88320);
    }

    return c;
}

int main() {
    unsigned char str[] = "correct horse battery staple";
    uint32_t crc = ~0;
    int i;

    for (i = 0; str[i] != '\0'; i++) {
        crc = crc32(crc, &str[i], 1);
    }

    printf("%" PRIu32 "\n", crc);

    return 0;
}
