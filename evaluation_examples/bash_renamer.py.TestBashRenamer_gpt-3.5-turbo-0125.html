#!/bin/bash

# Check if the user provided a directory path
if [ $# -ne 1 ]; then
    echo "Usage: $0 /path/to/files"
    exit 1
fi

# Get the directory path from the user input
directory=$1

# Check if the directory exists
if [ ! -d "$directory" ]; then
    echo "Directory does not exist"
    exit 1
fi

# Rename all files of the form "merged_*" to "finished_*"
for file in $directory/merged_*; do
    if [ -f "$file" ]; then
        new_file=$(echo "$file" | sed 's/merged_/finished_/')
        mv "$file" "$new_file"
        echo "Renamed $file to $new_file"
    fi
done

echo "All files renamed successfully"
