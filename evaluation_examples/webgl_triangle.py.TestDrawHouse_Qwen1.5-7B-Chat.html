// Set up the scene, camera, and renderer
const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('myCanvas') });

// Create a cube for the house
const houseGeometry = new THREE.BoxGeometry(50, 100, 50);
const houseMaterial = new THREE.MeshBasicMaterial({ color: 'gray' });
const house = new THREE.Mesh(houseGeometry, houseMaterial);

// Create a cylinder for the chimney
const chimneyGeometry = new THREE.CylinderGeometry(10, 10, 50, 10); // Adjust the dimensions as needed
const chimneyMaterial = new THREE.MeshBasicMaterial({ color: 'black' });
const chimney = new THREE.Mesh(chimneyGeometry, chimneyMaterial);

// Position the house and chimney
house.position.set(0, 50, 0);
chimney.position.set(0, 70, 0);

// Add objects to the scene
scene.add(house);
scene.add(chimney);

// Set camera and renderer

renderer.setSize(window.innerWidth, window.innerHeight);
