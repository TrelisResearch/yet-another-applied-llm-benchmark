use std::collections::HashMap;
use std::fs::File;
use std::io::{BufRead, BufReader};

fn main() {
    let file_path = "/tmp/file.txt";
    let file = File::open(file_path).expect("Unable to open file");
    let reader = BufReader::new(file);

    let mut word_count: HashMap<String, u32> = HashMap::new();

    for line in reader.lines() {
        if let Ok(line) = line {
            let words = line.split_whitespace();
            for word in words {
                *word_count.entry(word.to_string()).or_insert(0) += 1;
            }
        }
    }

    print_json(&word_count);
}

fn print_json(word_count: &HashMap<String, u32>) {
    print!("{{");
    let mut first = true;
    for (word, count) in word_count {
        if !first {
            print!(", ");
        } else {
            first = false;
        }
        print!("\"{}\": {}", word, count);
    }
    println!("}}");
}
To run this program, save it to a file such as `rust_word_count.rs` and compile it with `rustc rust_word_count.rs`. Then run the program by calling `./rust_word_count`. Make sure to have a text file located at `/tmp/file.txt` with the text you want to analyze. The program will output the word count in JSON format to stdout.
