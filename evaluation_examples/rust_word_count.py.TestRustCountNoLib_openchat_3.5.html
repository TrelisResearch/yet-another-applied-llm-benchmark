use std::collections::HashMap;
use std::fs::File;
use std::io::prelude::*;
use std::path::Path;

fn main() {
    let mut word_counts = HashMap::new();

    if let Some(file_path) = std::env::args().nth(1) {
        if let Ok(mut file) = File::open(&file_path) {
            for line in std::io::BufReader::new(file).lines() {
                let line = line.unwrap();
                for word in line.split_whitespace() {
                    *word_counts.entry(word).or_insert(0) += 1;
                }
            }
        }
    }

    let mut json = String::new();
    json.push('{');
    for (word, count) in word_counts {
        json.push_str(&format!("\"{}\": {}", word, count));
        json.push(',');
    }
    json.truncate(json.len() - 1); // Remove the trailing comma
    json.push('}');

    println!("{}", json);
}
To compile this program, save it to a file named `rust_word_count.rs` and run `rustc rust_word_count.rs`. Then you can run the compiled binary with `./rust_word_count /tmp/file.txt`.
