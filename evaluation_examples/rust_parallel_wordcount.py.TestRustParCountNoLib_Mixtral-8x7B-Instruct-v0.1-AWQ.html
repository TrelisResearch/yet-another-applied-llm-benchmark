use rayon::prelude::*;
use std::collections::HashMap;
use std::fs::File;
use std::io::{BufRead, BufReader, BufWriter, Write};
use std::path::Path;
use std::process;

fn main() {
    let input_path = "/tmp/file.txt";
    let file = File::open(input_path).expect("file should open readably");
    let reader = BufReader::new(file);

    let chunk_size = (1 << 20) * 4; // 4 MB chunks
    let chunks_count = num_cpus::get();
    let chunk_size_parts = (reader.count() + chunks_count as u64 - 1) / chunks_count as u64;

    let mut handles = Vec::new();
    let mut json_objs = Vec::new();

    for id in 0..chunks_count {
        let start = id * chunk_size_parts;
        let end = std::cmp::min((id + 1) * chunk_size_parts, reader.count());

        let json_obj = Arc::new(HashMap::new());
        let json_data = Arc::clone(&json_obj);

        let child = process::Command::new("sh")
            .arg("-c")
            .arg(&format!("head -n +{} {} | tail -n +{} | tr -s '[:paus:[:space:]]' '\\n' | tr '[:upper:]' '[:lower:]' | sort | uniq -c | awk '{{print $1\"\\\":\"$2\"\\", $1\"}}'",
                          (start + 1), input_path, (start + 1), (start + 1), (start + 1)))
            .stdout(process::Stdio::piped())
            .stderr(process::Stdio::null())
            .spawn()
            .expect("failed to start child process");

        let child_handler = rayon::join(
            move || {
                let mut child_stdin = child.stdin.lock();
                let input_str = &std::fs::read_to_string(&Path::new(input_path)).unwrap();
                let chunk = &input_str[start as usize..end as usize];
                child_stdin.write(chunk.as_bytes()).unwrap();
                let status = child.wait().unwrap_or_else(|e| e.into_code().unwrap_or(1));
                assert_eq!(status, 0, "failed to execute external process");
            },
            move || {
                let mut stdin = stdin();
                let mut buf = String::new();
                for line in stdin.lock().lines() {
                    buf.push_str(&line.unwrap_or_default());
                    let key_value: Vec<&str> = buf.splitn(2, ':').collect::<Vec<&str>>();
                    let word = key_value[0].trim();
                    let count = key_value[1].parse::<usize>().unwrap_or(0) + 1usize;
                    json_add_word(json_data, word, count);
                    buf.clear();
                }
            },
        );

        json_objs.push(json_obj.into_iter().collect::<HashMap<String, usize>>());
        handles.push(Box::new(child_handler));
    }

    for obj in json_objs.into_iter() {
        print!("{}", json::to_string_pretty(&obj).unwrap_or_default());
    }
