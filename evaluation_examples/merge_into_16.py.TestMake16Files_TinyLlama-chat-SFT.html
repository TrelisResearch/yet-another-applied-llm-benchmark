import os
import sys
import shutil

# Read the list of files from the command line
files = sys.argv[1]

# Create a list of file paths
file_paths = []
for file in files:
    file_paths.append(os.path.join(os.getcwd(), file))

# Merge the content of the files
merged_files = []
for i in range(16):
    merged_files.append(os.path.join(os.getcwd(), "merged_%d" % i))

# Write the merged files to disk
for file in merged_files:
    shutil.copyfile(file_paths[i], file)
