import math

def my_sqrt(a, tolerance=1e-6, max_iterations=1000):
    def f(x):
        return x**2 - a

    def f_prime(x):
        return 2 * x

    x0 = a
    x1 = x0
    iteration = 0

    while iteration < max_iterations:
        x1 = x0 - f(x0) / f_prime(x0)
        if abs(x1 - x0) < tolerance:
            break
        x0 = x1
        iteration += 1

    return x1

# Example usage:
