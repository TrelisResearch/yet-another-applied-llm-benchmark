#include <stdio.h>
#include <stdlib.h>

// BMP file header structure
typedef struct {
    uint16_t signature;      // 'B' for BMP, '5' for 5-bit color
    uint32_t fileSize;       // Size of the file in bytes
    uint16_t reserved1;      // Always 0
    uint16_t reserved2;      // Always 0
    uint32_t offsetData;     // Offset to pixel data in bytes
    uint32_t headerSize;     // Size of the header in bytes
    uint32_t width;          // Flag width
    uint32_t height;         // Flag height
    uint16_t planes;         // 1 (monochrome)
    uint16_t bitsPerPixel;   // 1 (5-bit)
    uint32_t compression;    // 0 (no compression)
    uint32_t imageSize;      // Size of the pixel data
    uint32_t xPixelsPerMeter; // 0 (not used)
    uint32_t yPixelsPerMeter; // 0 (not used)
    uint32_t colorsUsed;     // 1 (red)
    uint32_t colorsImportant; // 1 (red)
} BMPHeader;

// Pixel data for the Italian flag (red)
const uint8_t flag_pixels[] = {
    0xFF, 0x00, 0x00, // Red
    0x00, 0x00, 0xFF, // White
};

// Function to write BMP header and pixel data to stdout
void write_bmp(const BMPHeader& header, const uint8_t* pixels, int width, int height) {
    // Write BMP header
    printf("BM\n");
    printf("Size: %u\n", header.fileSize);
    printf("Width: %u\n", header.width);
    printf("Height: %u\n", header.height);
    printf("Planes: %u\n", header.planes);
    printf("BitCount: %u\n", header.bitsPerPixel);
    printf("Compression: %u\n", header.compression);
    printf("ImageSize: %u\n", header.imageSize);
    printf("XPixelsPerMeter: 0\n");
    printf("YPixelsPerMeter: 0\n");
    printf("ColorsUsed: %u\n", header.colorsUsed);
    printf("ColorsImportant: %u\n", header.colorsImportant);
    printf("\n");

    // Write pixel data
    for (int y = 0; y < height; ++y) {
        for (int x = 0; x < width; ++x) {
            printf("%02X", pixels[(height - y - 1) * width + x]);
        }
        printf("\n");
    }
}

int main() {
    BMPHeader header = {
        .signature = 0x424D, // BMP signature
        .fileSize = sizeof(BMPHeader) + sizeof(flag_pixels), // Total size
        .offsetData = sizeof(BMPHeader), // Offset to pixel data
        .headerSize = sizeof(BMPHeader),
        .width = 10, // Adjust this value for the flag size
        .height = 6, // Adjust this value for the flag size
        .planes = 1,
        .bitsPerPixel = 5, // 5-bit color
        .compression = 0,
        .imageSize = sizeof(flag_pixels),
        .colorsUsed = 1,
        .colorsImportant = 1,
    };

    // Write the flag pixels
    write_bmp(header, flag_pixels, header.width, header.height);

    return 0;
}
