use std::fs::File;
use std::io::{BufRead, BufReader};
use std::collections::HashMap;
use std::sync::{Arc, Mutex};
use std::thread;

fn count_words(filename: &str, start: usize, end: usize, result: Arc<Mutex<HashMap<String, usize>>>) {
    let file = File::open(filename).expect("Error opening file");
    let reader = BufReader::new(file);
    let mut word_count = HashMap::new();

    let mut current_line = 0;
    for line in reader.lines() {
        if let Ok(line) = line {
            if current_line >= start && current_line < end {
                for word in line.split_whitespace() {
                    *word_count.entry(word.to_string()).or_insert(0) += 1;
                }
            }
            current_line += 1;
        }
    }

    let mut result = result.lock().unwrap();
    *result = word_count;
}

fn main() {
    let filename = "/tmp/file.txt";
    let file = File::open(&filename).expect("Error opening file");
    let reader = BufReader::new(file);
    let lines: Vec<_> = reader.lines().collect();
    let total_lines = lines.len();
    let chunk_size = total_lines / 4;

    let result1 = Arc::new(Mutex::new(HashMap::new()));
    let result2 = Arc::clone(&result1);
    let result3 = Arc::clone(&result1);
    let result4 = Arc::clone(&result1);

    let handle1 = thread::spawn(move || {
        count_words(&filename, 0, chunk_size, result1);
    });

    let handle2 = thread::spawn(move || {
        count_words(&filename, chunk_size, chunk_size * 2, result2);
    });

    let handle3 = thread::spawn(move || {
        count_words(&filename, chunk_size * 2, chunk_size * 3, result3);
    });

    let handle4 = thread::spawn(move || {
        count_words(&filename, chunk_size * 3, total_lines, result4);
    });

    handle1.join().unwrap();
    handle2.join().unwrap();
    handle3.join().unwrap();
    handle4.join().unwrap();

    let result1 = result1.lock().unwrap();
    let result2 = result2.lock().unwrap();
    let result3 = result3.lock().unwrap();
    let result4 = result4.lock().unwrap();

    println!("{:?}", *result1);
    println!("{:?}", *result2);
    println!("{:?}", *result3);
    println!("{:?}", *result4);
}
