// Set up the scene, camera, and renderer
const canvas = document.getElementById('canvas');
const renderer = new THREE.WebGLRenderer();
renderer.setSize(window.innerWidth, window.innerHeight);
canvas.appendChild(renderer.domElement);

const camera = new THREE.PerspectiveCamera(75, canvas.width / canvas.height, 0.1, 1000);
camera.position.z = 5; // Set the camera position

// Create a scene
const scene = new THREE.Scene();

// Add a ground plane
const ground = new THREE.PlaneGeometry(100, 100, 100, 100);
const groundMaterial = new THREE.MeshBasicMaterial({ color: 0x888888 });
const groundMesh = new THREE.Mesh(ground, groundMaterial);
scene.add(groundMesh);

// Add a chimney
const chimneyGeometry = new THREE.BoxGeometry(10, 10, 20); // Chimney size
const chimneyMaterial = new THREE.MeshBasicMaterial({ color: 0x804040 }); // Chimney color
const chimney = new THREE.Mesh(chimneyGeometry, chimneyMaterial);
chimney.position.y = 10; // Place it above the ground
scene.add(chimney);

// Set up animation
function animate() {
    requestAnimationFrame(animate);
    renderer.render(scene, camera);
}

// Start the animation
animate();
