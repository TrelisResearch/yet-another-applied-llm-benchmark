from datetime import timedelta
import re

def parse_transcript(content):
    transcript_segments = []
    pattern = r'\[(\d{2}:\d{2}:\d{2}\.\d{3}) --> (\d{2}:\d{2}:\d{2}\.\d{3})\]  (.*)'
    matches = re.findall(pattern, content)

    for start_time, end_time, text in matches:
        start_seconds = timedelta(hours=int(start_time[:2]), minutes=int(start_time[3:5]), seconds=int(start_time[6:8]), milliseconds=int(start_time[9:])).total_seconds()
        end_seconds = timedelta(hours=int(end_time[:2]), minutes=int(end_time[3:5]), seconds=int(end_time[6:8]), milliseconds=int(end_time[9:])).total_seconds()
        transcript_segments.append((start_seconds, end_seconds, text))

    return transcript_segments

def merge(content):
    parsed_transcript = parse_transcript(content)
    sentences = []
    current_sentence = []

    for start_time, end_time, text in parsed_transcript:
        current_sentence.append((start_time, end_time, text))
        if text.endswith('.'):
            sentence = ' '.join(word[2] for word in current_sentence)
            sentences.append((start_time, end_time, sentence))
            current_sentence = []

    return [sentence[2] for sentence in sentences]
The `merge()` function first calls `parse_transcript()` to get a list of word segments with their start and end times. It then initializes an empty list `sentences` to store the merged sentences, and an empty list `current_sentence` to store the words in the current sentence. The function then iterates over the parsed transcript, and for each word, it appends it to `current_sentence`. If the word ends in a period, the function joins the words in `current_sentence` into a single string, and appends it to the `sentences` list. It then resets `current_sentence` to be empty. Finally, the function returns a list of the merged sentences as strings. Here's an example usage of the `merge()` function:
