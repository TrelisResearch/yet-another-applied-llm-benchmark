use std::fs::File;
use std::io::BufReader;
use std::io::BufRead;
use std::thread;
use std::sync::mpsc;
use std::collections::HashMap;

fn main() {
    let file_path = "/tmp/file.txt";
    let file = File::open(file_path).unwrap();
    let reader = BufReader::new(file);

    let (tx, rx) = mpsc::channel();

    let mut word_counts: HashMap<String, usize> = HashMap::new();

    for line in reader.lines() {
        let line = line.unwrap();
        let words: Vec<&str> = line.split_whitespace().collect();

        for word in words {
            let word = word.to_lowercase();
            *word_counts.entry(word).or_insert(0) += 1;
        }
    }

    let mut threads = vec![];
    for i in 0..4 {
        let (tx_clone, rx_clone) = tx.split();
        let word_counts_clone = word_counts.clone();
        threads.push(thread::spawn(move || {
            let mut word_counts_local = word_counts_clone;
            let mut line_count = 0;
            for line in reader.lines().skip(i * reader.lines().count() / 4) {
                let line = line.unwrap();
                let words: Vec<&str> = line.split_whitespace().collect();

                for word in words {
                    let word = word.to_lowercase();
                    *word_counts_local.entry(word).or_insert(0) += 1;
                }
                line_count += 1;
            }
            tx_clone.send((line_count, word_counts_local)).unwrap();
        }));
    }

    for _ in 0..4 {
        let (line_count, word_counts) = rx.recv().unwrap();
        println!("{{ \"{}\": {}, \"{}\": {}}", word_counts.keys().next().unwrap(), word_counts.values().next().unwrap(), line_count);
    }
}
This program uses the `mpsc` crate to send and receive messages between threads. It splits the file into 4 parts and runs each part in a separate thread. Each thread counts the number of lines and the number of occurrences of each word in its part of the file. The program then prints the JSON output for each thread. To run this program, you can compile it with `rustc yourfile.rs` and then run it with `./yourfile`.
