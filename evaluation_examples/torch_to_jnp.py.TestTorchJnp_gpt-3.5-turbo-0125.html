import jax
import jax.numpy as jnp

class MLP:
    def __init__(self, input_size, hidden_size, output_size):
        self.params = {
            'linear1': jnp.zeros((input_size, hidden_size)),
            'bias1': jnp.zeros(hidden_size),
            'linear2': jnp.zeros((hidden_size, output_size)),
            'bias2': jnp.zeros(output_size)
        }

    def linear(self, x, w, b):
        return jnp.dot(x, w) + b

    def relu(self, x):
        return jnp.maximum(x, 0)

    def softmax(self, x):
        exp_x = jnp.exp(x - jnp.max(x, axis=1, keepdims=True))
        return exp_x / jnp.sum(exp_x, axis=1, keepdims=True)

    def __call__(self, x):
        x = self.linear(x, self.params['linear1'], self.params['bias1'])
        x = self.relu(x)
        x = self.linear(x, self.params['linear2'], self.params['bias2'])
        return self.softmax(x)

model = MLP(10, 20, 30)
key = jax.random.PRNGKey(0)
x = jax.random.normal(key, (5, 10))
y = model(x)
print(list(y.flatten().sum(axis=1)))
