import sys
import struct

# Define the GIF header
GIF_HEADER = b'\x47\x49\x46\x38\x39\x61'  # GIF89a
GIF_WIDTH = 200
GIF_HEIGHT = 200
GIF_COLOR_COUNT = 2  # Red and white
GIF_LUMINANCE_SIZE = 1
GIF_COLOR_SIZE = 3
GIF_IMAGE_ID = 0
GIF_VERSION = 89

# Define the pixel data for the triangle
triangle_pixels = [
    [255, 0, 0],  # Red
    [255, 255, 255],  # White
    [0, 255, 0],  # Red (top left to bottom right)
]

# Write the GIF header
sys.stdout.write(GIF_HEADER)
sys.stdout.write(struct.pack('!2I', GIF_WIDTH, GIF_HEIGHT))

# Write the color palette (2 colors: red, white)
for i in range(GIF_COLOR_COUNT):
    color = (triangle_pixels[0][0], triangle_pixels[0][1], triangle_pixels[0][2])
    sys.stdout.write(struct.pack('!3B', *color))
    if i < GIF_COLOR_COUNT - 1:
        sys.stdout.write(struct.pack('!1B', 0))  # No transparency for white

# Write the image data (200x200 pixels)
for y in range(GIF_HEIGHT):
    for x in range(GIF_WIDTH):
        # Check if the pixel is part of the triangle
        if x == 0 or x == GIF_WIDTH - 1 or y == 0 or y == GIF_HEIGHT - 1:
            sys.stdout.write(struct.pack('!3B', *triangle_pixels[y][x]))
        else:
            sys.stdout.write(struct.pack('!3B', 255, 255, 255))  # White

# Write the GIF trailer
sys.stdout.write(struct.pack('!2B', 0x3B, 0x49))  # End of file, IEND
sys.stdout.write(struct.pack('!2I', 0, 0))  # Unknown, will be ignored
