use std::collections::HashMap;
use std::env;
use std::fs::File;
use std::io::{BufRead, BufReader};
use std::path::Path;
use std::sync::Arc;
use std::vec::Vec;
use crossbeam::thread;

fn main() {
    let args: Vec<String> = env::args().collect();
    if args.len() != 2 {
        eprintln!("Usage: {} <file>", args[0]);
        return;
    }

    let file_path = Path::new(&args[1]);
    let file = File::open(file_path).expect("Failed to open file");
    let reader = BufReader::new(file);

    let mut lines = reader.lines();
    let line_count = lines.count();
    let chunk_size = line_count / 4;

    let mut threads = Vec::new();
    let mut counts = Vec::new();

    for i in 0..4 {
        let start = i * chunk_size;
        let end = if i == 3 { line_count } else { (i + 1) * chunk_size };
        let lines_iter = lines.skip(start).take(end - start);

        let handle = thread::scope(|s| {
            let mut counts_arc = Arc::new(HashMap::new());
            s.spawn(move || {
                let mut counts = counts_arc.clone();
                for line in lines_iter {
                    let line = line.expect("Failed to read line");
                    let words: Vec<&str> = line.split_whitespace().collect();
                    for word in words {
                        let count = counts.entry(word.to_owned()).or_insert(0);
                        *count += 1;
                    }
                }
            });
            counts_arc
        });

        counts.push(handle.join().unwrap());
    }

    for (i, counts_arc) in counts.into_iter().enumerate() {
        let mut counts = counts_arc.into_inner();
        let json_str = serde_json::to_string(&counts).unwrap();
        println!("{{\"thread-{}\": {}}}", i, json_str);
    }
}
This program reads the input file line by line and splits it into 4 equal chunks. Each chunk is processed in parallel by a separate thread. The `crossbeam` crate is used to manage the threads. The program then merges the counts from each thread and prints the final result as JSON. To compile the program, run `rustc -C target-feature=+crt-static -o rust_word_count yourfile.rs`. To run the program, call `./rust_word_count /path/to/file.txt`. Note: The `serde_json` crate is used to serialize the final result as JSON. If you don't want to use an external crate, you can implement the JSON serialization manually, but it will be more complex and error-prone.
