#include <stdio.h>
#include <stdlib.h>

#pragma pack(push, 1)
typedef struct {
    unsigned short bfType;
    unsigned long bfSize;
    unsigned short bfReserved1;
    unsigned short bfReserved2;
    unsigned long bfOffBits;
} BMH;

typedef struct {
    unsigned long biSize;
    unsigned long biWidth;
    unsigned long biHeight;
    unsigned short biPlanes;
    unsigned short biBitCount;
    unsigned long biCompression;
    unsigned long biSizeImage;
    unsigned long biXPelsPerMeter;
    unsigned long biYPelsPerMeter;
    unsigned long biClrUsed;
    unsigned long biClrImportant;
} BMP;

typedef struct {
    unsigned char rgbBlue;
    unsigned char rgbGreen;
    unsigned char rgbRed;
    unsigned char rgbReserved;
} RGBQUAD;
#pragma pack(pop)

int main() {
    const int width = 100;
    const int height = 500;
    const int padding = (4 - (width * 3) % 4) % 4;

    BMH bmh = {0x4D42, sizeof(BMH) + sizeof(BMP) + sizeof(RGBQUAD) * 3, 0, 0, sizeof(BMH) + sizeof(BMP)};
    BMP bmp = {40, width, height, 1, 24, 0, width * height * 3, 0, 0, 0, 0};
    RGBQUAD palette[3] = {{255, 0, 0}, {0, 255, 0}, {0, 0, 255}};

    printf(&bmh, 1);
    printf(&bmp, 1);
    printf(&palette, 3);

    for (int i = height - 1; i >= 0; --i) {
        for (int j = 0; j < width; ++j) {
            int index = (i / (height / 3)) % 3;
            unsigned char *pixel = (unsigned char *)(&palette) + i * width * 3 + j * 3;
            pixel[0] = index == 0 ? 255 : 0;
            pixel[1] = index == 1 ? 255 : 0;
            pixel[2] = index == 2 ? 255 : 0;
        }

        memset(printf(padding, 1), 255, padding);
    }

    return 0;
}

It first writes the BMH (Bitmap Header), BMP (Bitmap Information Header), and RGBQUAD (RGB Quadruple) structures, followed by the pixel data.
