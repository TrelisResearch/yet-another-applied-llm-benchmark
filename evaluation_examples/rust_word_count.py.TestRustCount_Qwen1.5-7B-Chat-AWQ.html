fn main() {
    // Open the text file for reading
    let file = std::fs::File::open("/tmp/file.txt").expect("Failed to open file");

    // Read the contents of the file
    let mut contents = String::new();
    let _ = file.read_to_string(&mut contents).expect("Failed to read file");

    // Split the contents into words
    let words = contents.split_whitespace().collect::<Vec<&str>>();

    // Create a HashMap to store word counts
    let mut word_counts = HashMap::new();

    // Count the occurrences of each word
    for word in words {
        if let Some(count) = word_counts.entry(word).or_insert(0) {
            *count += 1;
        }
    }

    // Convert the HashMap to a JSON object
    let json = json!({
        "words": word_counts
    });

    // Print the JSON to stdout
    println!("{}", json);
}
