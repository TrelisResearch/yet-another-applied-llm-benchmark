#include <arpa/inet.h>
#include <stdio.h>

#define CRC32_INIT 0xFFFFFFFF

// Function to perform XOR operation on 32-bit integers
uint32_t xor32(uint32_t a, uint32_t b) {
    return (a ^ b);
}

// Function to calculate CRC32 hash
uint32_t calculate_crc32(const char *str, int len) {
    uint32_t crc = CRC32_INIT;
    for (int i = 0; i < len; i++) {
        crc = xor32(crc, (uint32_t)str[i]);
    }
    return crc;
}

int main() {
    const char *str = "correct horse battery staple";
    int len = strlen(str);

    uint32_t crc = calculate_crc32(str, len);
    printf("CRC32 hash: %08X\n", crc);

    return 0;
}
