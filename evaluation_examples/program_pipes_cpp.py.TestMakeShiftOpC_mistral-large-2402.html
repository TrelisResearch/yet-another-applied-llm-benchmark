#include <iostream>

class Const {
public:
    explicit Const(int value) : value_(value) {}

    Const& operator>> (const Square& sq) {
        sq.operate(value_);
        return *this;
    }

private:
    int value_;
};

class Square {
public:
    void operate(int& value) const {
        value = value * value;
        Inc inc;
        inc.operate(value);
        Print print;
        print.operate(value);
    }
};

class Inc {
public:
    void operate(int& value) const {
        value = value + 1;
    }
};

class Print {
public:
    void operate(const int& value) const {
        std::cout << value << std::endl;
    }
};

Const Cost(int value) {
    return Const(value);
}

int main() {
    Cost(5) >> Square();
    return 0;
}
In this code, `Cost(5)` creates an instance of the `Const` class with the value 5. The `>>` operator is overloaded to call the `operate` method of the `Square` class, which squares the value and then calls the `operate` method of the `Inc` class to increment the value. Finally, the `operate` method of the `Print` class is called to print the value to the screen.
