use std::collections::HashMap;
use std::env;
use std::fs::File;
use std::io::{self, prelude::*, BufRead};
use std.io::Lines;

fn main() -> io::Result<()> {
    let args: Vec<String> = env::args().collect();
    if args.len() != 2 {
        eprintln!("Usage: {} FILENAME", args[0]);
        std::process::exit(1);
    }

    let filename = &args[1];
    let mut word_count: HashMap<String, u32> = HashMap::new();

    // Read the specified file
    let file = File::open(filename)?;
    let reader = io::BufReader::new(file);

    // Count the occurrence of each word
    for line in reader.lines() {
        let line = line?;
        for word in line.split_whitespace() {
            let word = word.to_lowercase();
            let count = word_count.entry(word).or_insert(0);
            *count += 1;
        }
    }

    // Serialize the HashMap to JSON and print the result
    let json = serde_json::to_string_pretty(&word_count).unwrap();
    println!("{}", json);

    Ok(())
}

use std::collections::HashMap;
use std::env;
use std::fs::File;
use std::io::{self, prelude::*, BufRead, BufReader};
use std::io::ErrorKind;
use serde_json;
use serde_json::Value;

fn read_words_from_file(filename: &str) -> Result<HashMap<String, u32>, io::Error> {
    let file = File::open(filename)?;
    let reader = BufReader::new(file);

    let mut word_count: HashMap<String, u32> = HashMap::new();

    for line in reader.lines() {
        let line = line?;
        for word in line.split_whitespace() {
            let word = word.to_lowercase();
            let count = word_count.entry(word).or_insert(0);
            *count += 1;
        }
    }

    Ok(word_count)
}

fn main() -> Result<io::Result<()>, &'static str> {
    let args: Vec<String> = env::args().collect();
    if args.len() != 2 {
        eprintln!("Usage: {} FILENAME", args[0]);
        std::process::exit(1);
    }

    let filename = &args[1];

    let word_count_map = read_words_from_file(filename)?;

    let json = serde_json::to_vec_pretty(&word_count_map)?;

    println!("{}", String::from_utf8_unchecked(&json));

    Ok(())
}
