import jax.numpy as jnp
from jax import jit, vmap
from jax.nn import one_hot

@jit
def one_hot(indices, num_classes):
    return one_hot(indices, num_classes)

# If you want to apply this function to a list of indices, you can use vmap
# vmap creates a vectorized version of the function which can take a list of indices
one_hot_vectorized = vmap(one_hot)
