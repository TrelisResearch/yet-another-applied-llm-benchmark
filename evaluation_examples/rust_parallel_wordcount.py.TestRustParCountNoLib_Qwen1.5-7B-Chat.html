use std::env;
use std::fs;
use std::io;
use std::thread;

fn main() {
    // Get the file path from the command line
    let file_path = env::args().next().unwrap_or("/tmp/file.txt");

    // Open the file in read mode
    let file = fs::File::open(file_path).expect("Failed to open file");

    // Calculate the chunk size (1/4 of the file)
    let chunk_size = (file.metadata().unwrap().len() as f64) / 4.0;

    // Create a vector to store the word counts
    let mut word_counts = vec![HashMap::new(); 4];

    // Read the file in chunks and count words
    let mut buffer = String::new();
    for chunk in file.chunks(chunk_size) {
        // Process the chunk in a separate thread
        let _ = thread::spawn(|&mut word_counts, chunk| {
            count_words_in_chunk(chunk, &mut word_counts);
        }).join();

        // Drain the buffer to process the current chunk
        buffer.extend_from_slice(chunk);
        count_words_in_buffer(&buffer, &mut word_counts);
        buffer.clear();
    }

    // Print the JSON output for each thread's word counts
    for (i, counts) in word_counts.iter().enumerate() {
        println!("{{\"word\": {}, \"count\": {}}}", i + 1, counts.len());
        for (word, count) in counts.iter() {
            println!("    \"{}\": {}", word, count);
        }
        println!();
    }
}

fn count_words_in_chunk(chunk: &[u8], word_counts: &mut Vec<HashMap<String, u32>>) {
    let mut words = chunk.iter().flat_map(|c| c.split_whitespace().cloned());
    for word in words {
        if !word.is_empty() {
            word_counts[i].insert(word.to_string(), 1);
        }
    }
}

fn count_words_in_buffer(buffer: &str, word_counts: &mut Vec<HashMap<String, u32>>) {
    for word in buffer.split_whitespace() {
        if !word.is_empty() {
            word_counts.iter_mut().find(|&mut wc| wc.insert(word.to_string(), 1)).unwrap();
        }
    }
}
