function move(x) {
  const moves = [];

  for (let i = 1; i < x.length; i++) {
    for (let j = 0; j < x.length - i; j++) {
      const substring = x.substring(j, j + i);
      const newX = x.substring(0, j) + substring + x.substring(j + i);
      moves.push(newX);
    }
  }

  return moves;
}
