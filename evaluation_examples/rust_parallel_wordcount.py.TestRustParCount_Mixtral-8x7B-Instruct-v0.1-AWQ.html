use rayon::prelude::*;
use rayon::slice;
use std::collections::HashMap;
use std::env;
use std::fs::File;
use std::io::prelude::*;
use std::path::Path;

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();
    if args.is_empty() {
        eprintln!("Usage: {} /path/to/input.txt", env::current_exe().unwrap().to_string_lossy());
        return;
    }

    let input_path = Path::new(&args[0]);
    let input_str = {
        let mut file = File::open(&input_path).expect("Failed to read the input file");
        let mut input_str = String::new();
        file.read_to_string(&mut input_str)
            .expect("Failed to read the entire file as a single string");
        input_str
    };

    let chunk_size = input-str.len() / 4;
    let chunks = input_str.as_str().granulate_threaded(chunk_size, 4);

    let thread_count = num_cpus::get();
    let thread_chunk_size = chunks.len() / thread_count;

    let mut thread_results: HashMap<String, u32> = HashMap::new();

    rayon::join(
        (0..thread_chunk_size).chain(..),
        (0..thread_chunk_size).chain(..),
        (0..thread_chunk_size).chain(..),
        (0..thread_chunk_size).chain(..),
        move || {
            for (i, chunk) in chunks.skip(thread_chunk_size * rayon::current_index_range().next_index()).take(thread_chunk_size).enumerate() {
                let mut word_frequencies = HashMap::new();
                for word in chunk.split_whitespace() {
                    let word = word.to_lowercase();
                    let count = word_frequencies.entry(word).or_insert(0);
                    *count += 1;
                }
                for (word, count) in word_frequencies.iter() {
                    let entry = thread_results.entry(word.to_owned()).or_insert(0);
                    *entry += count;
                }
            }
        },
    );

    let mut json_output: String = "{\n";
    let mut first = true;
    for (word, count) in thread_results.iter() {
        if !first {
            json_output.push_str(",\n");
        }
        json_output.push_str(&format!("\"{word}\": {count}"));
        first = false;
    }
    json_output.push_str("\n}\n");

    println!("{}", json_output.replace("\\u003e", "\\\\u003e"));
}

use rayon::slice;
use std::collections::HashMap;
use std::env;
use std::fs::File;
use std::io::prelude::*;
use std::path::Path;
use std::sync::mpsc_queue::MPSCQueue;
use std::sync::Arc;
use std::thread;

fn parallel_word_count(chunk: &str, queue: Arc<MPSCQueue<((String, u32, u32, u32), u32)>>) {
    let mut word_frequencies = HashMap::new();
    for word in chunk.split_whitespace() {
        let word = word.to_lowercase();
        let count = word_frequencies.entry(word).or_insert(0);
        *count += 1;
    }
    for (word, count) in word_frequencies.iter() {
        let entry = queue.send((*word, 1, *count, chunk.len() as u32)).unwrap();
        entry.wait().unwrap();
    }
}
